                    ago_time += (text == "day") ? ((numberOfUnits < 10) ? "0" + numberOfUnits : numberOfUnits) + " " + text + ((numberOfUnits > 1) ? "s " : " ") : ((numberOfUnits < 10) ? "0" + numberOfUnits : numberOfUnits) + ":";
                }
                document.getElementById("server_uptime").innerHTML = "UpTime&nbsp;-&nbsp;" + ago_time.slice(0, -1) + " ago";
                currentTimeStamp++;
            }

	function getAgentinfo() {
